#!/bin/bash
su - ec2-user << 'EOF'
export AWS_ACCESS_KEY=<AWS ACCESS KEY>
export AWS_SECRET_KEY=<AWS SECRET KEY>
export $DOCKER_IMAGE=<docker image to pull>
export NXF_VER=<version>

EOF
